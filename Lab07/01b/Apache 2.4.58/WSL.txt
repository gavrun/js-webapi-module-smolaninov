sudo apt update

sudo apt install -y apache2

sudo mkdir -p /var/www/lab071b
sudo cp -r <project>/* /var/www/lab071b
sudo chown -R www-data:www-data /var/www/lab071b

sudo tee /etc/apache2/sites-available/lab071b.conf >/dev/null <<'EOF'
<VirtualHost *:8081>
    ServerName localhost
    DocumentRoot /var/www/lab071b

    <Directory /var/www/lab071b>
        Options Indexes FollowSymLinks
        AllowOverride All
        Require all granted
    </Directory>

</VirtualHost>
EOF

echo "Listen 8081" | sudo tee -a /etc/apache2/ports.conf

sudo a2ensite lab071b

sudo systemctl reload apache2
# sudo systemctl restart apache2

http://localhost:8081/index.html
